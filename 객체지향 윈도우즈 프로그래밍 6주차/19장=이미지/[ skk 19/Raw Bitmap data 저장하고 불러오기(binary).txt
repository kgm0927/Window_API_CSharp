		private void saveBinaryRaw(... )
		{
			FileStream fileStream = new FileStream(fileName, FileMode.Create, FileAccess.Write);
			BinaryWriter binWr = new BinaryWriter(fileStream);

			binWr.Write(bitmap.Width);
			binWr.Write(bitmap.Height);
			for(int i = 0; i < bitmap.Height; i++) {
				for(int j = 0; j < bitmap.Width; j++) {
					Color color = bitmap.GetPixel(j, i);

					binWr.Write(color.R);
					binWr.Write(color.G);
					binWr.Write(color.B);
				}
			}

			fileStream.Flush();
			fileStream.Close();
		}
